public Optional<NicknameResult> getPlayerNickname(String id, NicknameType type) {

    String key = id + ":" + type;

    // 1. Check cache
    if (cache.containsKey(key)) {
        return Optional.of(cache.get(key));
    }

    // 2. Fetch player
    Optional<Player> playerOpt = playerRepository.findById(id);
    if (playerOpt.isEmpty()) {
        return Optional.empty();
    }

    Player player = playerOpt.get();

    // 3. Select generator
    NicknameGenerator generator =
            (type == NicknameType.AI) ? aiGenerator : ruleGenerator;

    // 4. Generate nickname
    NicknameResult result = generator.generate(player);

    // 5. Store in cache
    cache.put(key, result);

    return Optional.of(result);
}

public Players searchWithPagination(
        String birthstate,
        String bats,
        Integer experience,
        Integer page,
        Integer size) {

    // default values
    if (page == null || page < 0) {
        page = 0;
    }
    if (size == null || size <= 0) {
        size = 20;
    }

    // 1. Filter first
    Players filteredPlayers = search(birthstate, bats, experience);
    List<Player> list = filteredPlayers.getPlayers();

    int listSize = list.size();

    // 2. Compute pagination
    int start = page * size;

    if (start >= listSize) {
        return new Players(); // empty result
    }

    int end = Math.min(start + size, listSize);

    List<Player> pagedList = list.subList(start, end);

    Players result = new Players();
    result.setPlayers(pagedList);

    return result;
}

“This approach works for small datasets, but for large datasets I’d push pagination to the database using Pageable.”

    public Players getTopPlayersUsingHeap(int n) {
        if (n <= 0) {
            return new Players();
        }

        PriorityQueue<Player> minHeap = new PriorityQueue<>((a,b)->Integer.compare(parseHeightSafely(a.getHeight()),parseHeightSafely(b.getHeight())));
        List<Player> players = playerRepository.findAll();
        for(Player p:players){
            minHeap.add(p);
